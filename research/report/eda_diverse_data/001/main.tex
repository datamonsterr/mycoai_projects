\section{Target Structure}
\begin{verbatim}

            └── {Strain}_{angle}_{id}_seg{n}.jpg   (segments)
└── diverse_data_metadata.json
\end{verbatim}

\section{JSON Metadata Schema}
Each image entry includes:
\begin{itemize}
\item \texttt{id}: Unique identifier
\item \texttt{file\_path}: Path to preprocessed image
\item \texttt{bboxes\_image}: Path to image with bounding boxes
\item \texttt{data}: Object containing species, strain, environment, angle
\item \texttt{bboxes}: Array of $\{xmin, ymin, xmax, ymax\}$ objects
\item \texttt{num\_colonies}: Number of colonies detected
\item \texttt{segment\_paths}: Array of paths to segmented colony images
\end{itemize}
