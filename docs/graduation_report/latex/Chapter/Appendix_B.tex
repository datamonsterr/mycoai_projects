\section{Use Case Specifications}

\subsection{UC-RETRIEVE-01: Species Retrieval from Segmented Dish Image}

\begin{table}[ht]
\centering
\caption{UC-RETRIEVE-01: Species Retrieval from Segmented Dish Image}
\label{tab:uc_retrieve}
\begin{tabularx}{\textwidth}{@{}p{3cm} X@{}}
\toprule
\textbf{Field} & \textbf{Description} \\
\midrule
Actor & Research mycologist or lab technician \\
Description & User uploads a Petri dish image; system segments colonies, extracts features, queries Qdrant, and returns ranked species predictions \\
Precondition & User is authenticated; at least one reference collection exists in Qdrant \\
Postcondition & Top-K species predictions with confidence scores and nearest-neighbor evidence displayed \\
Priority & High \\
Frequency & Several times per lab session \\
Normal Course &
\begin{enumerate}[nosep]
  \item User navigates to Retrieve Species page
  \item User uploads a Petri dish image (PNG/JPEG, up to 20MB)
  \item System segments the dish into individual colony crops using K-Means + HSV refinement
  \item User reviews and optionally adjusts bounding boxes
  \item System extracts feature embeddings from each crop using EfficientNetB1 (fine-tuned)
  \item System queries Qdrant for K nearest neighbors per crop
  \item System aggregates scores across crops using freq\_strength strategy
  \item System applies threshold formula (gnorm\_0\_2) for known/unknown classification
  \item System displays ranked species list with confidence scores
\end{enumerate} \\
Alternative Courses &
\begin{enumerate}[nosep]
  \item \textbf{YOLO segmentation}: If YOLO model is available, user can switch to learned segmentation
  \item \textbf{Manual bounding box}: User can draw/correct colony bounding boxes before extraction
  \item \textbf{Adjust K}: User can change K from default 11 to 3, 5, 7, or 9
  \item \textbf{Upload batch}: User can upload a ZIP of multiple dish images for batch processing
\end{enumerate} \\
Exceptions &
\begin{enumerate}[nosep]
  \item \textbf{No colonies detected}: System reports ``No colonies found'' and suggests manual segmentation
  \item \textbf{Qdrant unavailable}: System displays error ``Reference database temporarily unavailable''
  \item \textbf{All colonies classified unknown}: System displays ``No confident match found'' with nearest neighbor gallery
  \item \textbf{Image too large}: Automatic resize to 256$\times$256
\end{enumerate} \\
Special Requirements & Response time < 5 seconds for up to 10 colonies with K=11 \\
Assumptions & Reference database is pre-populated with well-lit, in-focus colony images \\
\bottomrule
\end{tabularx}
\end{table}

\subsection{UC-DATA-01: Dataset Management \& Reference Indexing}

\begin{table}[ht]
\centering
\caption{UC-DATA-01: Dataset Management \& Reference Indexing}
\label{tab:uc_data}
\begin{tabularx}{\textwidth}{@{}p{3cm} X@{}}
\toprule
\textbf{Field} & \textbf{Description} \\
\midrule
Actor & Lab administrator or researcher \\
Description & Administrator manages the reference dataset — uploads new reference images, triggers embedding extraction, and indexes into Qdrant \\
Precondition & User has admin role; reference images are properly labeled with species name \\
Postcondition & New reference data is available for retrieval queries \\
Priority & Medium \\
Frequency & Weekly to monthly (when new reference strains are added) \\
Normal Course &
\begin{enumerate}[nosep]
  \item Admin uploads reference images with species label metadata
  \item System segments images using current segmentation pipeline
  \item System extracts feature embeddings from each segment
  \item System indexes embeddings into Qdrant with metadata (strain ID, species, growth medium, angle)
  \item Dashboard reflects updated dataset statistics
\end{enumerate} \\
Alternative Courses &
\begin{enumerate}[nosep]
  \item \textbf{Bulk import}: Admin uses script to batch-process a directory of labeled images
  \item \textbf{Replace collection}: Admin replaces entire collection for a new experiment
\end{enumerate} \\
Exceptions &
\begin{enumerate}[nosep]
  \item \textbf{Duplicate strain}: System warns if a strain ID already exists
  \item \textbf{Unlabeled image}: System rejects images without species metadata
\end{enumerate} \\
Special Requirements & Indexing of 1,000 images should complete within 30 minutes \\
Assumptions & Reference images follow the same imaging protocol as query images \\
\bottomrule
\end{tabularx}
\end{table}
