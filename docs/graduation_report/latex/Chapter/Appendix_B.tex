\section{Use Case Specifications}

\subsection{UC-RETRIEVE-01: Species Retrieval from Segmented Dish Image}

\begin{table}[H]
\centering
\caption{UC-RETRIEVE-01: Species Retrieval from Segmented Dish Image}
\label{tab:uc_retrieve}
\small
\begin{tabularx}{\textwidth}{@{}p{3cm} X@{}}
\toprule
\textbf{Field} & \textbf{Description} \\
\midrule
Primary Actor & User, Data Owner \\
Goal & Predict fungal species from uploaded strain images \\
Includes & UC-PREP-01: Prepare Segmented Images \\
Preconditions & Actor is authenticated; reference vectors indexed in Qdrant \\
Postconditions & Ranked species predictions displayed with confidence scores and KNN neighbor evidence; batch CSV downloadable; actor may submit feedback \\
Priority & Must \\
Stability & Stable \\
Normal Course &
\begin{enumerate}[nosep]
  \item Actor uploads one image or a batch folder to Retrieve page
  \item Actor provides strain and media metadata (species not required for retrieval)
  \item System invokes UC-PREP-01: auto-segmentation $\rightarrow$ bounding-box review
  \item Actor edits/confirms bounding boxes
  \item System selects KNN strategy: same-media for managed Media, all-media for new/other Media
  \item System extracts feature vector from each segment using active feature extractor
  \item System queries Qdrant for $K$ nearest neighbors per segment (cosine similarity)
  \item System aggregates votes across $N$ segments using configured aggregation strategy
  \item System applies threshold formula to classify prediction as known or unknown
  \item System displays ranked species list with confidence scores
\end{enumerate} \\
Alternative Courses &
\begin{enumerate}[nosep]
  \item \textbf{New/other Media}: System uses all-media KNN, flags Media for Data Owner review
  \item \textbf{Batch upload}: Actor downloads Batch Template Folder for structure guidance
  \item \textbf{Poor segmentation}: Actor manually adjusts bounding boxes before processing
  \item \textbf{Removed image}: System skips removed image
  \item \textbf{No valid segment}: System blocks retrieval until at least one segment exists
\end{enumerate} \\
Exceptions &
\begin{enumerate}[nosep]
  \item \textbf{Invalid upload structure}: System rejects batch and shows template mismatch
  \item \textbf{Qdrant unavailable}: System displays error 503 ``Vector search currently unavailable''
  \item \textbf{Image too small}: System rejects images below minimum dimensions
  \item \textbf{All colonies unknown}: System displays ``No confident match found''
\end{enumerate} \\
Special Requirements & Single-image retrieval response $<$ 5 seconds; batch progress visible within 2 seconds \\
Assumptions & Qdrant is pre-populated with at least one reference collection \\
\bottomrule
\end{tabularx}
\end{table}

\subsection{UC-DATA-01: Index New Data}

\begin{table}[H]
\centering
\caption{UC-DATA-01: Index New Data}
\label{tab:uc_data}
\small
\begin{tabularx}{\textwidth}{@{}p{3cm} X@{}}
\toprule
\textbf{Field} & \textbf{Description} \\
\midrule
Primary Actor & Data Owner \\
Goal & Add known reference data to retrieval database \\
Includes & UC-PREP-01: Prepare Segmented Images; UC-META-01: Manage Metadata \\
Preconditions & Data Owner authenticated; Species and Media metadata exist or are created during workflow \\
Postconditions & Reference images/segments are indexed or staged for Qdrant re-index; audit log recorded \\
Priority & Must \\
Stability & Stable \\
Normal Course &
\begin{enumerate}[nosep]
  \item Data Owner chooses single or batch indexing
  \item Data Owner uploads images or structured batch folder
  \item Data Owner supplies required metadata: species, strain, media
  \item Data Owner reviews images, metadata, and bounding boxes (UC-PREP-01)
  \item System auto-segments uploaded images
  \item Data Owner adjusts/confirms bounding boxes
  \item Data Owner removes unwanted data before submit
  \item System stores accepted images/segments as reference data
  \item System indexes data into Qdrant or marks it for re-indexing
  \item System records audit log
\end{enumerate} \\
Alternative Courses &
\begin{enumerate}[nosep]
  \item \textbf{Missing Species/Media}: Data Owner creates metadata through UC-META-01
  \item \textbf{Batch indexing}: System reads species metadata from folder structure/template
  \item \textbf{Accepted contribution feedback}: System presents pending reference data for Data Owner review
  \item \textbf{Existing duplicate}: System warns and requires Data Owner decision
\end{enumerate} \\
Exceptions &
\begin{enumerate}[nosep]
  \item \textbf{Duplicate strain}: System warns if strain ID already exists
  \item \textbf{Unlabeled image}: System rejects images without species metadata
\end{enumerate} \\
Special Requirements & Batch indexing of 1,000 images should complete within 30 minutes \\
Assumptions & Reference images follow the same imaging protocol as query images \\
\bottomrule
\end{tabularx}
\end{table}
